#Register User
Method: POST
URL: http://localhost:3000/api/auth/register

#Under JSON
{
  "email": "testuser@example.com",
  "password": "password123",
  "name": "Test User",
  "student_id": "2025001",
  "phone_number": "0978123456"
}

#Expected outcome
{
  "message": "User registered successfully",
  "user": {
    "user_id": "...",
    "email": "testuser@example.com",
    "name": "Test User"
  },
  "token": "eyJhbGciOi..."
}

------------------------------------------------

#Login
Method: POST
URL: http://localhost:3000/api/auth/login

{
  "email": "testuser@example.com",
  "password": "password123"
}

#Expected outcome
{
  "message": "Login successful",
  "user": {
    "user_id": "...",
    "email": "testuser@example.com",
    "name": "Test User"
  },
  "token": "eyJhbGciOi..."
}

-------------------------------------------------

#Get Profile
Method: GET
URL: http://localhost:3000/api/auth/profile

#Headers
Key: Authorization
Value: Bearer <paste_token_here>

#Expected outcome
{
  "user_id": "...",
  "email": "testuser@example.com",
  "name": "Test User",
  "student_id": "2025001",
  "phone_number": "0978123456"
}

---------------------------------------------
